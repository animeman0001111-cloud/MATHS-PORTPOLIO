Experiment 1 — Pixels as Vectors

Loading an image as a NumPy array, extracting RGB pixel vectors, and computing difference/magnitude between neighbouring pixels.

experiment1.py
import numpy as np
import matplotlib.pyplot as plt
from PIL import Image

img_pil = Image.open("sunset.jpeg").convert("RGB")
img = np.array(img_pil)
print("image shape (rows, cols, channels):", img.shape)

plt.imshow(img); plt.title("Original Image"); plt.axis("off"); plt.show()

# Access a specific pixel
i, j = 50, 80
pixel1 = img[i, j].reshape(3,1)
print("Pixel RGB Vector:\n", pixel1)

R, G, B = pixel1[0], pixel1[1], pixel1[2]
print("Red:", R, "Green:", G, "Blue:", B)

# Compare neighbouring pixel
pixel2 = img[i+1, j].reshape(3,1)
difference = pixel2 - pixel1
magnitude = np.linalg.norm(difference)
print("Difference vector:\n", difference)
print("Change magnitude:", magnitude)

Experiment 3 — Image Transformations (Scale, Rotate, Shear, Project)

Applying linear geometric transformations to an image: scaling, rotation, shearing, projection, and comparing composition order (scale→rotate vs rotate→scale).

experiment3.py
from PIL import Image
import matplotlib.pyplot as plt

img = Image.open("P1.jpeg")
print("Original size:", img.size)   # (389, 389)

# Scaling
scale = 4
scaled_img = img.resize((img.size[0]*scale, img.size[1]*scale))
print("Scaled size:", scaled_img.size)   # (1556, 1556)

# Rotation
rotated_img = img.rotate(45, fillcolor=(0,0,0))

# Multiple rotations
for angle in [30, 60, 90, 120]:
    img.rotate(angle, fillcolor=(0,0,0))

# Shearing
shear_factor = 0.5
w,h = img.size
shear_matrix = (1, shear_factor, 0, 0, 1, 0)
sheared_img = img.transform(
    (int(w + shear_factor*h), h),
    Image.AFFINE, shear_matrix, fillcolor=(0,0,0)
)

# Projection (vertical squashing)
projected_img = img.resize((w, int(h*0.1)))

# Order of operations: scale then rotate vs rotate then scale
scale, angle = 2, 45
scaled = img.resize((w*scale, h*scale))
scale_then_rotate = scaled.rotate(angle, fillcolor=(0,0,0))

rotated = img.rotate(angle, fillcolor=(0,0,0))
rotate_then_scale = rotated.resize((rotated.size[0]*scale, rotated.size[1]*scale))
Output
Original size: (389, 389)
Scaled size:   (1556, 1556)

→ Rotation rotates image about its center, expanding canvas (black fill)
→ Shearing skews the image along x-axis, widening the bounding box
→ Projection compresses image vertically (10% of original height)
→ Scale→Rotate and Rotate→Scale produce visually similar results,
  but pixel-level order of operations differs slightly due to
  interpolation at each step (non-commutativity of transforms
  on discrete pixel grids)

Experiment 4 — Matrix Transformations on Images (Brightness & Contrast)

Treating image pixels as vectors and applying matrix multiplication (contrast matrix) and vector addition (brightness vector), including repeated application via matrix powers.

experiment4.py
import numpy as np
import matplotlib.pyplot as plt
from PIL import Image

img_pil = Image.open("P1.jpeg")
img = np.array(img_pil)[:,:,:3]

I = np.eye(3)                       # Identity matrix
C = np.array([[1.5,0,0],[0,1.5,0],[0,0,1.5]])   # Contrast matrix
b = np.array([40,40,40])            # Brightness vector

bright_img    = img + b
composed_img  = np.tensordot(img, C.T, axes=1) + (C @ b)
contrast_img  = np.tensordot(bright_img, C.T, axes=1)

# Repeated contrast application via matrix powers
C2 = np.linalg.matrix_power(C,2)
C3 = np.linalg.matrix_power(C,3)
contrast_twice  = np.tensordot(img, C2.T, axes=1)
contrast_thrice = np.tensordot(img, C3.T, axes=1)

def clip(x): return np.clip(x,0,255).astype(np.uint8)

titles = ["Original","Brightness","Matrix Composition",
          "Contrast(C)","Contrast Twice(C^2)","Contrast Thrice(C^3)"]
images = [clip(img), clip(bright_img), clip(composed_img),
          clip(contrast_img), clip(contrast_twice), clip(contrast_thrice)]

plt.figure(figsize=(14,8))
for i,im in enumerate(images):
    plt.subplot(2,3,i+1); plt.imshow(im); plt.title(titles[i]); plt.axis("off")
plt.show()
Output
→ Brightness vector b shifts all pixel values up (image lightens)
→ Contrast matrix C = 1.5·I scales pixel intensities (contrast increases)
→ Composed transform = Contrast applied to shifted vector (Ax + b form)
→ C^2 and C^3 (repeated contrast) progressively oversaturate
  the image, clipping to pure white at higher powers

Experiment 5 — Solving Linear Systems (Gaussian Elimination, LU Decomposition)

Custom Gaussian elimination solver with RREF, rank-based classification (unique / infinite / no solution), parametric solutions using SymPy, and LU decomposition for production-planning systems.

experiment5.py
from sympy import symbols, Matrix
import sympy as sp

def solve_system(A, b):
    EPS = 1e-9
    m, n = len(A), len(A[0])
    M = [A[i][:] + [b[i]] for i in range(m)]

    # Forward elimination with partial pivoting
    row = 0
    for col in range(n):
        if row >= m: break
        max_row = max(range(row,m), key=lambda r: abs(M[r][col]))
        if abs(M[max_row][col]) < EPS: continue
        M[row], M[max_row] = M[max_row], M[row]
        for r in range(row+1, m):
            ratio = M[r][col]/M[row][col]
            for c in range(col, n+1):
                M[r][c] -= ratio*M[row][c]
        row += 1

    # Check consistency
    for i in range(m):
        if all(abs(M[i][j])EPS:
            print("No solution"); return

    pivot_cols, pivot_rows = [], []
    for i in range(m):
        for j in range(n):
            if abs(M[i][j]) > EPS:
                pivot_rows.append(i); pivot_cols.append(j); break
    rank = len(pivot_cols)

    if rank < n:
        print("Infinite solutions")
        free_vars = [j for j in range(n) if j not in pivot_cols]
        t = symbols('t0:'+str(len(free_vars)))
        sol = [0]*n
        for i,fv in enumerate(free_vars): sol[fv] = t[i]
        for i in reversed(range(rank)):
            r,c = pivot_rows[i], pivot_cols[i]
            s = M[r][n]
            for j in range(c+1,n): s -= M[r][j]*sol[j]
            sol[c] = s/M[r][c]
        for i in range(n): print(f"x{i+1} =", sol[i])
    else:
        x = [0]*n
        for i in reversed(range(rank)):
            r,c = pivot_rows[i], pivot_cols[i]
            s = M[r][n]
            for j in range(c+1,n): s -= M[r][j]*x[j]
            x[c] = s/M[r][c]
        print("Unique solution:")
        for i in range(n): print(f"x{i+1} =", round(x[i],6))

# 10x10 system (infinite solutions, rank 9)
A = [[5,1,1,0,1,0,0,1,0,1],[1,6,1,1,0,1,0,0,1,0], ...]
b = [60,70,50,80,45,30,40,35,55,60]
solve_system(A, b)

# --- LU Decomposition (production plan) ---
sp.init_printing()
A = Matrix(A); b = Matrix(b)
L, U, perm = A.LUdecomposition()
P = sp.eye(A.shape[0])
for i,j in enumerate(perm): P.row_swap(i,j)

y = L.solve(P*b)
x = U.solve(y)   # fails if U is singular

Experiment 6 — Rank, Column Space, Null Space & Recommendation Logic

Using SymPy to compute RREF, rank, column space, and null space of a user–genre matrix, interpreting redundancy to drive a Netflix-style personalization recommendation.

experiment6.py
import sympy as sp

def analyze_matrix(A):
    A = sp.Matrix(A)
    print("Matrix:\n", A)

    R, pivots = A.rref()
    print("RREF:\n", R)

    rank = len(pivots)
    print("Rank:", rank)

    print("Column Space Basis:")
    for c in A.columnspace(): print(c)

    print("Null Space Basis:")
    for n in A.nullspace(): print(n)

    num_features = A.shape[1]
    redundancy = num_features - rank
    print("Total Features:", num_features)
    print("Independent Features (Rank):", rank)
    print("Redundant Features:", redundancy)

    # Genre interpretation
    genres = ["Action","Horror","Sci-Fi"]
    if rank == 0:
        print("No data — Default/Trending content, Personalization: None")
    if rank == 1:
        print("All genres behave the same — Personalization: Very Low")
    elif rank == len(genres):
        print("All genres independent — Personalization: High")
        for i,row in enumerate(A.tolist()):
            idx = row.index(max(row))
            print(f"User {i+1} → More {genres[idx]}")
    else:
        print("Partial redundancy — Personalization: Medium")
        for i,row in enumerate(A.tolist()):
            idx = row.index(max(row))
            print(f"User {i+1} → Likely prefers {genres[idx]}")
        if len(A.nullspace()) > 0:
            print("Some behavior changes are invisible (null space present)")
Output (conceptual interpretation)
→ rank = 0  →  No pattern detected → Personalization: None
→ rank = 1  →  All genres collapse to one direction
               → Same recommendation for every user → Personalization: Very Low
→ rank = 3  →  Action, Horror, Sci-Fi fully independent
               → Unique per-user recommendation → Personalization: High
→ 0 < rank < 3 → Partial redundancy
               → Medium personalization; null space indicates
                 hidden user-behavior dimensions Netflix cannot see

Experiment 7 — Least Squares Fitting & QR Projection (Movie Recommendation)

Projecting a user's preference vector onto a "movie matrix" using QR decomposition and the normal equation, then ranking individual movies by least-squares error to pick the best fit.

experiment7.py
import numpy as np
import matplotlib.pyplot as plt

def analyze_user(U, movies):
    A = np.column_stack(list(movies.values()))
    Q, R = np.linalg.qr(A)

    U_proj = Q @ Q.T @ U                      # QR projection
    x = np.linalg.inv(A.T @ A) @ (A.T @ U)    # normal equation
    y_fit = A @ x
    lse = np.linalg.norm(U - y_fit)**2

    print("Projected:", np.round(U_proj).astype(int))
    print("Fitted values (Ax):", np.round(y_fit).astype(int))
    print("LSE:", round(lse,2))

    plt.plot(U,'o-',label='Original')
    plt.plot(y_fit,'s--',label='Fitted(Ax)')
    plt.legend(); plt.show()

    best, err = None, float('inf')
    for n, v in movies.items():
        p = ((U @ v)/(v @ v)) * v
        e = np.linalg.norm(U - p)**2
        print(n, "LSE:", round(e,2))
        if e < err: err, best = e, n
    print("Top recommended movie:", best)

movies = {
    "M1_Tower Heist": np.array([1,0,1]),
    "M2_Stree": np.array([0,1,1]),
    "M3_Army of the dead": np.array([1,1,0])
}

# Case 1: balanced strong user
analyze_user(np.array([7,6,1]), movies)

# Case 3: conflicting / horror-dominant user
analyze_user(np.array([0,5,1]), movies)

Experiment 8 — Eigenvalues, Eigenvectors & Genre Recommendation

Computing eigenvalues/eigenvectors of a user-similarity matrix, classifying trend strength (dominant/balanced/weak), and recommending a genre via matrix-vector multiplication.

experiment8.py
import numpy as np

def recommend_genre(A, users, g):
    vals, vecs = np.linalg.eig(A)
    print("Eigenvalues:", np.round(vals,2))
    print("Eigenvectors:\n", np.round(vecs,2))

    for v in vals:
        if v > 1:   print(v, "Strong dominant trend / important pattern")
        elif v == 1: print(v, "Natural or balanced trend")
        else:        print(v, "Weak or less significant trend")

    print("\nRecommendations:")
    for n, u in users.items():
        u = np.array(u)
        rec = g[np.argmax(A @ u)]
        print(n, "Recommended genre:", rec)

genres = ["Action","Horror","Sci-fi"]
A = np.array([[2,1,1],
              [1,2,1],
              [1,1,2]])

# Case 1: Balanced (eigen-aligned) user
users = {"User1": [5,5,5]}
recommend_genre(A, users, genres)
